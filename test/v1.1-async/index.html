Default:


<div id="DC-note-42282" class="DC-note-container"></div>
<script src="loader.js"></script>
<script>
  dc.embed.loadNote('https://www.documentcloud.org:443/documents/282753-lefler-thesis/annotations/42282.js', {
    container: '#DC-note-42282'
  });
</script>
